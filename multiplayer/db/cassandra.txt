CREATE TABLE users (
    login text,
    password text,
    money decimal,
    items list,
    PRIMARY KEY (login)
 );

CREATE TABLE chat (
  id int,
  users set,
  messages list,
  PRIMARY KEY (id)
);

CREATE TABLE server (
   id int,
   quantity int,
   PRIMARY KEY(id)
);

CREATE TABLE groups (
   name text,
   users set,
   count int,
   PRIMARY KEY(name)
);
