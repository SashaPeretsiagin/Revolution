CREATE TABLE chat (
  id int,
  users set,
  messages list,
  PRIMARY KEY (id)
);

CREATE TABLE server (
   id int,
   quantity int,
   PRIMARY KEY(id)
);

CREATE TABLE groups (
   name text,
   users set,
   count int,
   PRIMARY KEY(name)
);

CREATE TABLE ratings (
rank text,
users set,
PRIMARY KEY(rank)
)
CREATE TABLE notifications (
id int,
users set,
PRIMARY KEY(id)
);
